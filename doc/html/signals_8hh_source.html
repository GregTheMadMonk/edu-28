149 runInBulkHelper(std::size_t bulkSize, Func func, Args... args)
150 {
151 const auto threads = std::thread::hardware_concurrency();
152
153 using ResultType = std::invoke_result_t<Func, Args...>;
154
155 std::vector<ResultType> ret(bulkSize);
156 std::vector<std::thread> workers;
157
158 std::size_t start = 0;
159 for (std::size_t thread = 0; thread < threads; ++thread) {
160 std::size_t end = (thread == threads - 1) ? bulkSize : (start + bulkSize / threads);
161
162 workers.emplace_back(
163 [start, end, &ret, &func, &args...] {
164 for (std::size_t i = start; i < end; ++i) {
165 ret[i] = std::invoke(func, args...);
166 }
167 }
168 );
169
170 start = end;
171 }
172
173 for (auto& t : workers) t.join();

175 return ret;
176 } // <-- runInBulkHelper()
